52/**
53 * Show the graph in the middle of the view, scaled to fit
54 */
55function show()
56{
57 if (window.innerHeight) // Firefox
58 {
59 windowWidth = window.innerWidth;
60 windowHeight = window.innerHeight;
61 }
62 else if (document.documentElement.clientWidth) // Chrome/Safari
63 {
64 windowWidth = document.documentElement.clientWidth
65 windowHeight = document.documentElement.clientHeight
66 }
67 if (!windowWidth || !windowHeight) // failsafe
68 {
69 windowWidth = 800;
70 windowHeight = 600;
71 }
72 minZoom = Math.min(Math.min(viewHeight,windowHeight)/viewHeight,Math.min(viewWidth,windowWidth)/viewWidth);
73 maxZoom = minZoom+1.5;
74 zoomInFactor = Math.pow(maxZoom/minZoom,1.0/zoomSteps);
75 zoomOutFactor = 1.0/zoomInFactor;
76
77 var g = svgDoc.getElementById('viewport');
78 try
79 {
80 var bb = g.getBBox(); // this can throw an exception if css { display: none }
81 var tx = (windowWidth-viewWidth*minZoom+8)/(2*minZoom);
82 var ty = viewHeight+(windowHeight-viewHeight*minZoom)/(2*minZoom);
83 var a = 'scale('+minZoom+') rotate(0) translate('+tx+' '+ty+')';
84 g.setAttribute('transform',a);
85 }
86 catch(e) {}
87}

160function doZoom(g,point,zoomFactor)
161{
162 var p = point.matrixTransform(g.getCTM().inverse());
163 var k = root.createSVGMatrix().translate(p.x, p.y).scale(zoomFactor).translate(-p.x, -p.y);
164 var n = g.getCTM().multiply(k);
165 var s = Math.max(n.a,n.d);
166 if (s>maxZoom) n=n.translate(p.x,p.y).scale(maxZoom/s).translate(-p.x,-p.y);
167 else if (s<minZoom) n=n.translate(p.x,p.y).scale(minZoom/s).translate(-p.x,-p.y);
168 setCTM(g, n);
169 stateTf = stateTf.multiply(n.inverse());
170}

172/**
173 * Handle mouse move event.
174 */
175function handleMouseWheel(evt)
176{
177 if (!evt) evt = window.evt;
178 if (!evt.shiftKey) return; // only zoom when shift is pressed
179 if (evt.preventDefault) evt.preventDefault();
180 evt.returnValue = false;
181
182 if (state!='pan')
183 {
184 var delta;
185 if (evt.wheelDelta)
186 {
187 delta = evt.wheelDelta / 7200; // Opera/Chrome/IE9/Safari
188 }
189 else
190 {
191 delta = evt.detail / -180; // Mozilla
192 }
193 var svgDoc = evt.target.ownerDocument;
194 var g = svgDoc.getElementById("viewport");
195 var p = getEventPoint(evt);
196 doZoom(g,p,1+delta);
197 }
198}

200/**
201 * Handle mouse move event.
202 */
203function handleMouseMove(evt)
204{
205 if(evt.preventDefault)
206 evt.preventDefault();
207
208 evt.returnValue = false;
209
210 var g = svgDoc.getElementById("viewport");
211
212 if (state == 'pan')
213 {
214 // Pan mode
215 var p = getEventPoint(evt).matrixTransform(stateTf);
216 setCTM(g,stateTf.inverse().translate(p.x - stateOrigin.x, p.y - stateOrigin.y));
217 }
218}

220/**
221 * Handle click event.
222 */
223function handleMouseDown(evt)
224{
225 if(evt.preventDefault)
226 evt.preventDefault();
227 evt.returnValue = false;
228 var g = svgDoc.getElementById("viewport");
229 state = 'pan';
230 stateTf = g.getCTM().inverse();
231 stateOrigin = getEventPoint(evt).matrixTransform(stateTf);
232 g.style.cursor = cursorGrab;
233}

235/**
236 * Handle mouse button release event.
237 */
238function handleMouseUp(evt)
239{
240 if (evt.preventDefault) evt.preventDefault();
241 evt.returnValue = false;
242 var g = svgDoc.getElementById("viewport");
243 g.style.cursor = "default";
244 // Quit pan mode
245 state = '';
246}

288function serializeXmlNode(xmlNode)
289{
290 if (typeof window.XMLSerializer != "undefined") {
291 return (new window.XMLSerializer()).serializeToString(xmlNode);
292 } else if (typeof xmlNode.xml != "undefined") {
293 return xmlNode.xml;
294 }
295 return "";
296}
